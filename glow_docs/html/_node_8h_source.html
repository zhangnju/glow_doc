337  bool hasSameParent() const {
338  assert(it_ != getEnd() && "Cannot check invalid iterator");
339  // A users-list should be for one node.
340  // If this assert breaks, that means the input list is broken,
341  // or this iterator is not used as it was intended: to walk
342  // through a users-list.
343  assert(it_->get()->getNode() == parent_.getNode() &&
344  "Iterator points to a list with different parent?!");
345  return it_->get()->getResNo() == parent_.getResNo();
346  }

349  NodeValueIteratorImpl(NodeValueTy &parent, iterator it)
350  : parent_(parent), it_(it) {
351  if (it_ != getEnd() && !hasSameParent()) {
352  ++(*this);
353  }
354  assert((it_ == getEnd() || hasSameParent()) &&
355  "operator++ should return the next valid iterator");
356  }

359  NodeValueIteratorImpl &operator++() {
360  auto endIt = getEnd();
361  while (++it_ != endIt && !hasSameParent()) {
362  }
363  return *this;
364  }
