455  bool isEqual(const Type &other, bool allowDifferentShape = false) const {
456  // Element type must be the same.
457  if (elementType_ != other.elementType_) {
458  return false;
459  }
460  // Must have the same number of sizes.
461  if (numSizes_ != other.numSizes_) {
462  return false;
463  }
464  // Sizes must be the same.
465  if (!allowDifferentShape) {
466  for (size_t i = 0; i < numSizes_; i++) {
467  if (sizes_[i] != other.sizes_[i]) {
468  return false;
469  }
470  }
471  }
472 
473  // Strides must be the same.
474  for (size_t i = 0; i < numSizes_; i++) {
475  if (strides_[i] != other.strides_[i]) {
476  return false;
477  }
478  }
479 
480  // Compare the scale and offset of integers.
481  if (isQuantizedType()) {
482  if (scale_ != other.scale_ || offset_ != other.offset_) {
483  return false;
484  }
485  }
486 
487  return true;
488  }

654  void initDims(llvm::ArrayRef<size_t> dims,
655  llvm::ArrayRef<size_t> alignments) {
656  assert(dims.size() <= max_tensor_dimensions && "Too many dimensions.");
657  assert(dims.size() == alignments.size() &&
658  "The number of dimensions and alignments should be the same");
659  // Update the tensor strides and sizes based on given dims and alignments.
660  // Sizes are simply assigned to dims. And strides are computed as partial
661  // product of dims, making sure that each dimension is aligned as required.
662  numSizes_ = dims.size();
663  if (numSizes_ > 0) {
664  // Stride of the last dimension is always 1.
665  assert(alignments[numSizes_ - 1] == 1 &&
666  "Last dimension must always be aligned.");
667  strides_[numSizes_ - 1] = 1;
668  sizes_[numSizes_ - 1] = dims[numSizes_ - 1];
669  }
670  for (int i = numSizes_ - 2; i >= 0; i--) {
671  size_t alignment = alignments[i];
672  if (alignment != 1) {
673  assert(alignment % getElementSize() == 0 &&
674  "Alignment should be a multiple of element size");
675  alignment /= getElementSize();
676  }
677  // All the strides (except for last one) depend on the previous dimension.
678  strides_[i] = alignedSize(dims[i + 1] * strides_[i + 1], alignment);
679  assert(dims[i] > 0 && "Do not allow a dimension of zero.");
680  sizes_[i] = dims[i];
681  }
682  }
683 };
