52 inline Node *convertConvToNCHWConv(ConvolutionNode *CN, Function *F) {
53  // Convert filter and input from NHWC (Glow's default) into NCHW.
54  auto *NI = F->createTranspose("conv.input", CN->getInput(), NHWC2NCHW);
55  auto *NF = F->createTranspose("conv.filter", CN->getFilter(), NHWC2NCHW);
56 
57  auto dimsNHWC = ShapeNHWC(CN->getResult().getType()->dims());
58  auto dimsNCHW = {dimsNHWC.n, dimsNHWC.c, dimsNHWC.h, dimsNHWC.w};
59  auto outTy = F->getParent()->uniqueTypeWithNewShape(CN->getResult().getType(),
60  dimsNCHW);
61 
62  auto *NC = F->addNode(new ConvolutionNode(
63  CN->getName(), outTy, NI, NF, CN->getBias(), CN->getKernels(),
64  CN->getStrides(), CN->getPads(), CN->getGroup(), CN->getDilation(), NCHW,
65  CN->getFusedActivation()));
66  auto *NR = F->createTranspose("conv.result", NC, NCHW2NHWC);
67 
68  return NR;
69 }
