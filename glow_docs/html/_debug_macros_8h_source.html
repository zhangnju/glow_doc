24 // Macro for memory instrumentation.
25 #if NNPI_COLLECT_MEM_USAGE
26 #include <atomic>
27 #include <fstream>
28 #include <iostream>
29 #include <unistd.h>

31 static std::atomic<double> dbg_mem_usage_last_vm_value(0.0);
32 static std::atomic<double> dbg_mem_usage_last_rss_value(0.0);
33 #define DBG_MEM_USAGE(msg) \
34  { \
35  double vmSize = 0.0; \
36  double rssSize = 0.0; \
37  unsigned long vmInput; \
38  long rssInput; \
39  { \
40  std::string notUsed; \
41  std::ifstream ifs("/proc/self/stat", std::ios_base::in); \
42  for (int i = 0; i < 22 /* unused fields in stat line */; i++) \
43  ifs >> notUsed; \
44  ifs >> vmInput >> rssInput; \
45  } \
46  double pageSize = sysconf(_SC_PAGE_SIZE) / (1024.0 * 1024.0); \
47  vmSize = vmInput / (1024.0 * 1024.0); \
48  rssSize = rssInput * pageSize; \
49  LOG(INFO) << "[MEM_USAGE][" << __FUNCTION__ << "]" << msg \
50  << " [MEMORY] vm:" << static_cast<uint64_t>(vmSize) \
51  << " MB RSS:" << static_cast<uint64_t>(rssSize) << " MB " \
52  << "[DELTA] d_vm:" \
53  << static_cast<int64_t>(vmInput - dbg_mem_usage_last_vm_value) \
54  << " B D_RSS:" \
55  << static_cast<int64_t>(rssInput - dbg_mem_usage_last_rss_value) \
56  << " B \n"; \
57  dbg_mem_usage_last_vm_value = vmInput; \
58  dbg_mem_usage_last_rss_value = rssInput; \
59  }
60 #warning "#### DBG_MEM_USAGE ENABLED #####"
61 #else // Not NNPI_COLLECT_MEM_USAGE
62 #define DBG_MEM_USAGE(msg)
63 #endif // NNPI_COLLECT_MEM_USAGE
