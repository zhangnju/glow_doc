121  void zero(bool resetFusedScalesOffsets = false) {
122  size_t size = actualSize();
123  // Quantized tensors should go to their offset.
124  switch (type_.getElementType()) {
125  case ElemKind::Int8QTy: {
126  auto *data = reinterpret_cast<int8_t *>(getData());
127  std::fill(&data[0], &data[0] + size, (int8_t)type_.getOffset());
128  break;
129  }
130  case ElemKind::UInt8QTy: {
131  auto *data = reinterpret_cast<uint8_t *>(getData());
132  std::fill(&data[0], &data[0] + size, (uint8_t)type_.getOffset());
133  break;
134  }
135  case ElemKind::Int16QTy: {
136  auto *data = reinterpret_cast<int16_t *>(getData());
137  std::fill(&data[0], &data[0] + size, (int16_t)type_.getOffset());
138  break;
139  }
140  case ElemKind::Int32QTy: {
141  auto *data = reinterpret_cast<int32_t *>(getData());
142  std::fill(&data[0], &data[0] + size, (int32_t)type_.getOffset());
143  break;
144  }
145 #define FUSED_CASE(ELEM_KIND, DATA_TYPE) \
146  case ElemKind::ELEM_KIND: { \
147  assert(dims().size() == 2 && "Fused tensor must be 2-dimensional."); \
148  assert(dims()[1] > sizeof(DATA_TYPE) && \
149  "Fused tensor must have space for scale and offset."); \
150  const size_t dataWidth = dims()[1]; \
151  const size_t alignedLength = type_.strides()[0]; \
152  auto *data = reinterpret_cast<uint8_t *>(getData()); \
153  for (size_t i = 0, e = dims()[0]; i < e; i++) { \
154  uint8_t *scaleOffsetPtr = \
155  data + i * alignedLength + dataWidth - 2 * sizeof(DATA_TYPE); \
156  DATA_TYPE scale, offset; \
157  if (resetFusedScalesOffsets) { \
158  /* Use these as defaults, and copy them into each row. */ \
159  scale = 1.0; \
160  offset = 0.0; \
161  memcpy(scaleOffsetPtr, &scale, sizeof(DATA_TYPE)); \
162  memcpy(scaleOffsetPtr + sizeof(DATA_TYPE), &offset, \
163  sizeof(DATA_TYPE)); \
164  } else { \
165  memcpy(&scale, scaleOffsetPtr, sizeof(DATA_TYPE)); \
166  memcpy(&offset, scaleOffsetPtr + sizeof(DATA_TYPE), \
167  sizeof(DATA_TYPE)); \
168  } \
169  DCHECK_NE(static_cast<float>(scale), 0.0) \
170  << "Disallow scale = 0.0 for Fused ElemKinds; causes div by zero."; \
171  float zero = nearbyintf(-1 * static_cast<float>(offset / scale)); \
172  std::fill(data + i * alignedLength, scaleOffsetPtr, \
173  static_cast<uint8_t>(zero)); \
174  } \
175  break; \
176  }
177  FUSED_CASE(UInt8FusedQTy, float);
178  FUSED_CASE(UInt8FusedFP16QTy, float16_t);
179 #undef FUSED_CASE
180 
181  default:
182  // Non-quantized tensors are set to 0.
183  std::fill(&getData()[0], &getData()[0] + size * type_.getElementSize(),
184  0);
185  break;
186  }
187  }

281  llvm::ArrayRef<size_t> offsets = {}) const {

287  "Number of dims of tensor must equal number of dims in offsets");
288  // Find the index of the first element and use it to find the pointer to
289  // the first element.

292  index += type_.strides()[i] * offsets[i];

442  bool isEqualImpl(const Tensor &other, bool isBitwise, float allowedError,
443  bool verbose) const {
444  if (other.dims() != dims()) {
445  return false;
446  }
447 
448  // For now, make sure that either both or neither of the tensors have
449  // UInt8FusedQTy or UInt8Fused16QTy. While it is possible for an Int8QTy
450  // tensor to equal a fused tensor if the fused tensor has the same
451  // scale/offset on all of its rows, and that scale/offset match that of the
452  // Int8QTy, we do not support checking this for now.
453  assert(((getElementType() == ElemKind::UInt8FusedQTy &&
454  other.getElementType() == ElemKind::UInt8FusedQTy) ||
455  (getElementType() == ElemKind::UInt8FusedFP16QTy &&
456  other.getElementType() == ElemKind::UInt8FusedFP16QTy) ||
457  (getElementType() != ElemKind::UInt8FusedFP16QTy &&
458  other.getElementType() != ElemKind::UInt8FusedQTy)) &&
459  "Fused ElemKinds only supports comparing against same ElemKind.");
460 
461  // Assert that the scale and offset match for the quantized types.
462  switch (getElementType()) {
463  default:
464  break;
465  case ElemKind::Int8QTy:
466  case ElemKind::UInt8QTy:
467  case ElemKind::Int16QTy:
468  case ElemKind::Int32QTy:
469  assert(getType().getScale() == other.getType().getScale() &&
470  "Scales must match.");
471  assert(getType().getOffset() == other.getType().getOffset() &&
472  "Offsets must match.");
473  }
474 
475  // Bitwise compare.
476  if (isBitwise)
477  return isBitwiseEqualImpl(other);
478 
479  switch (getElementType()) {
480  case ElemKind::FloatTy:
481  return isEqualImpl<float>(other, allowedError, verbose);
482  case ElemKind::Float16Ty:
483  return isEqualImpl<float16_t>(other, allowedError, verbose);
484  case ElemKind::Int8QTy:
485  return isEqualImpl<int8_t>(other, allowedError, verbose);
486  case ElemKind::UInt8QTy:
487  return isEqualImpl<uint8_t>(other, allowedError, verbose);
488  case ElemKind::Int16QTy:
489  return isEqualImpl<int16_t>(other, allowedError, verbose);
490  case ElemKind::Int32QTy:
491  return isEqualImpl<int32_t>(other, allowedError, verbose);
492  case ElemKind::Int32ITy:
493  return isEqualImpl<int32_t>(other, allowedError, verbose);
494  case ElemKind::Int64ITy:
495  return isEqualImpl<int64_t>(other, allowedError, verbose);
496  // Note: We can use isEqualImpl() here because the scales/offsets will be
497  // compared as if they were data, so we will return false if any rowwise
498  // scale/offset do not match.
499  case ElemKind::UInt8FusedQTy:
500  return isEqualImpl<uint8_t>(other, allowedError, verbose);
501  case ElemKind::UInt8FusedFP16QTy:
502  return isEqualImpl<uint8_t>(other, allowedError, verbose);
503  case ElemKind::UInt4FusedFP16QTy:
504  return isEqualImpl<uint8_t>(other, allowedError, verbose);
505  case ElemKind::BoolTy:
506  return isEqualImpl<bool>(other, allowedError, verbose);
507  }
508 
509  // This is to make compiler happy. It can never reach this point as switch
510  // always covers all possible values.
511  llvm_unreachable("unreachable");
512  }

636  template <class ElemTy>
637  bool isEqualImpl(const Tensor &other, float allowedError,
638  bool verbose) const {
639  auto const *myData = getRawDataPointer<ElemTy>();
640  auto const *otherData = other.getRawDataPointer<ElemTy>();
641  double maxFoundError = 0.0;
642  size_t maxFoundErrorIdx = 0, numExceedingError = 0;
643  for (size_t i = 0, e = size(); i < e; i++) {
644  double delta = myData[i] - otherData[i];
645  delta = std::abs(delta);
646  // Since any comparison with NAN returns false, we use a negated condition
647  // so that this function correctly returns false when delta is NAN.
648  if (!(delta <= allowedError)) {
649  if (!verbose) {
650  return false;
651  }
652  numExceedingError += 1;
653  if (!(delta <= maxFoundError)) {
654  maxFoundError = delta;
655  maxFoundErrorIdx = i;
656  }
657  }
658  }
659  if (numExceedingError != 0) {
660  LOG(INFO) << "Tensors not equal: " << numExceedingError << " out of "
661  << size() << " elements exceeded allowed error threshold "
662  << allowedError << ". Maximum error found was " << maxFoundError
663  << " at index " << maxFoundErrorIdx << ": "
664  << myData[maxFoundErrorIdx] << " vs. "
665  << otherData[maxFoundErrorIdx];
666  }
667  return numExceedingError == 0;
668  }

1019  assert(dims().size() == 2 && "Fused tensor must be 2-dimensional."); \

1021  "Fused tensor must have space for scale/offset."); \

1023  const size_t alignedLength = tensor_->getType().strides()[0]; \

1025  for (size_t i = 0, e = dims()[0]; i < e; i++) { \
1026  uint8_t *scaleOffsetPtr = \
1027  data + i * alignedLength + dataWidth - 2 * sizeof(DATA_TYPE); \
1028  DATA_TYPE scale, offset; \
1029  memcpy(&scale, scaleOffsetPtr, sizeof(DATA_TYPE)); \
1030  memcpy(&offset, scaleOffsetPtr + sizeof(DATA_TYPE), sizeof(DATA_TYPE)); \
1031  for (size_t j = 0, e = dataWidth - 2 * sizeof(DATA_TYPE); j < e; j++) { \
1032  float currVal = (at({i, j}) * (float)scale) + (float)offset; \
1033  if (std::abs(currVal) > allowedError) { \
1034  return false; \
1035  } \
1036  } \
1037  } \
1038  return true;

1141  size_t n = actualSize();

1143

1147  mean += raw({i});

1225  void insertTensorsImpl(llvm::MutableArrayRef<size_t> sliceCoor,
1226  llvm::MutableArrayRef<size_t> fusedCoor,
1227  Handle<ElemTy> &slice, bool isInsert,
1228  llvm::ArrayRef<size_t> offset, size_t count,
1229  size_t axis, unsigned d) {
1230  bool isDone = (d == slice.dims().size());
1231 
1232  if (isDone) {
1233  if (isInsert) {
1234  at(fusedCoor) = slice.at(sliceCoor);
1235  } else {
1236  slice.at(sliceCoor) = at(fusedCoor);
1237  }
1238  return;
1239  }
1240 
1241  // Only need to iterate over count if the current dimension d is equal to
1242  // the axis we're inserting over.
1243  const size_t countIters = (axis == d) ? count : 1;
1244  for (size_t c = 0; c < countIters; c++) {
1245  for (size_t i = 0, e = slice.dims()[d]; i < e; i++) {
1246  // Construct the coordinates for the slice and for the joint shape.
1247  // Add the 'offset' to the dimension that we concat the shapes on.
1248  sliceCoor[d] = i;
1249  // If this is the correct axis to insert multiple times then calculate
1250  // the additional offset to use.
1251  const size_t countAxisOffset = (axis == d) ? c * slice.dims()[d] : 0;
1252  fusedCoor[d] = i + offset[d] + countAxisOffset;
1253  insertTensorsImpl(sliceCoor, fusedCoor, slice, isInsert, offset, count,
1254  axis, d + 1);
1255  }
1256  }
