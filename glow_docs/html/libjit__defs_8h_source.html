112 inline int32_t libjit_scale_i32i8(int32_t input, int32_t pre, int32_t post,
113  int32_t scale, int32_t offset) {
114  // The operation x >> post is rounded down to negative infinity. To get to
115  // round-nearest we add (1 << (post - 1)) to the value prior to shifting.
116  // Rounding is performed only when shifting right (pos > 0).
117  int rtn = (post > 0) ? (1 << (post - 1)) : 0;
118 
119  // NOTICE: If your tests are failing because of signed integer overflow then
120  // this is a bug in the test and not in the program. You should make sure that
121  // the inputs to the operations do not overflow. The semantics of the
122  // quantization process is such that the result for values that fall out of
123  // range is undefined. The conversion procedure will only tolerate a few bits
124  // of overflow and the result will be clipped.
125  return ((((input >> pre) * scale) + rtn) >> post) + offset;
126 }
