211 template <class LoaderType, class OpType>
212 Error constantFoldInLoader(Function *F, LoaderType &tmpLoader,
213  LoaderType *loader, const OpType &op) {
214  PlaceholderBindings bindings;
215  std::vector<Tensor *> outTensors;
216  Module *mod = F->getParent();
217 
218  // Register the constant inputs to the current op with the constant folding
219  // loader.
220  for (unsigned i = 0; i < op.input_size(); i++) {
221  Constant *tmpConst = mod->getConstantByName(op.input(i));
222  RETURN_ERR_IF_NOT(tmpConst, "No constant found");
223  tmpLoader.nodeValueByName_[op.input(i)] = tmpConst->getOutput();
224  }
225 
226  // Using the loader to load the current operator.
227  RETURN_IF_ERR(tmpLoader.loadOperator(op));
228 
229  // To collect the folded outputs allocate and add save nodes to the folding
230  // function.
231  for (int i = 0; i < op.output_size(); i++) {
232  const auto &outputName = op.output(i);
233  NodeValue r;
234  ASSIGN_VALUE_OR_RETURN_ERR(r, tmpLoader.getNodeValueByName(outputName));
235  Placeholder *PH =
236  F->getParent()->createPlaceholder(r.getType(), outputName, false);
237  SaveNode *SN = F->createSave("save_" + outputName, r, PH);
238  auto *result = bindings.allocate(SN->getPlaceholder());
239  outTensors.push_back(result);
240  }
241 
242  // Evaluate the constant outputs using interpreter backend.
243  std::unique_ptr<Backend> backend(createBackend("Interpreter"));
244  CompilationContext cctx;
245  cctx.compMode = CompilationMode::Infer;
246  cctx.optimizationOpts.enableConstantFolding = false;
247  cctx.backendOpts.collectConstants = true;
248  RETURN_IF_ERR(executeConstantFunction(*backend, *F, bindings, cctx));
249 
250  // Using the graph output, place constant nodes in the original graph.
251  for (int i = 0; i < op.output_size(); i++) {
252  RETURN_IF_ERR(loader->createAndRegisterConstant(op.output(i),
253  std::move(*outTensors[i])));
254  }
255 
256  return Error::success();
257 }
