53  int32_t transform(int32_t input) {
54  // The operation x >> post is rounded down to negative infinity. To get to
55  // round-nearest we add (1 << (post - 1)) to the value prior to shifting.
56  // Rounding is performed only when shifting right (pos > 0).
57  int rtn = (post > 0) ? (1 << (post - 1)) : 0;
58  return ((((input >> pre) * scale) + rtn) >> post) + offset;
59  }
60 };

279 template <typename ScaleT, typename OffsetT, typename QP>
280 void tensorRowwiseQuantization(const Tensor &input, Tensor &output,
281  Tensor &scales, Tensor &offsets,
282  quantization::Schema schema) {
283  constexpr bool offsetIsFP = std::is_same<float, OffsetT>::value ||
284  std::is_same<float16_t, OffsetT>::value;
285  constexpr bool offsetIsInt32 = std::is_same<int32_t, OffsetT>::value;
286  static_assert((offsetIsInt32 && std::is_same<float, ScaleT>::value) ||
287  (offsetIsFP && std::is_same<ScaleT, OffsetT>::value),
288  "Invalid combination of Scale/Offset types.");
289 
290  const auto fDims = flattenCdr(input.dims());
291  Tensor finalIn = input.getUnowned({fDims.first, fDims.second});
292  Tensor finalOut = output.getUnowned({fDims.first, fDims.second});
293  ShapeHW idim(finalIn.dims());
294 
295  auto srcH = finalIn.getHandle<float>();
296  auto destH = finalOut.getHandle<QP>();
297  auto scalesH = scales.getHandle<ScaleT>();
298  auto offsetsH = offsets.getHandle<OffsetT>();
299  for (size_t i = 0; i < idim.height; i++) {
300  auto slice = srcH.extractSlice(i);
301  auto rSrc = slice.getHandle<float>();
302  auto res = rSrc.minMaxArg();
303  float min = rSrc.raw(res.first);
304  float max = rSrc.raw(res.second);
305  // Expand the range to include 0.0f so that 0 is exactly representable.
306  min = std::min(min, 0.0f);
307  max = std::max(max, 0.0f);
308 
309  // Handle rowwise quantization for FCs.
310  if (offsetIsInt32) {
311  TensorQuantizationParams qParams =
312  chooseQuantizationParams(min, max, schema);
313  for (size_t j = 0; j < idim.width; j++) {
314  destH.at({i, j}) = quantization::quantize(srcH.at({i, j}), qParams);
315  }
316  scalesH.raw(i) = qParams.scale;
317  offsetsH.raw(i) = qParams.offset;
318  } else if (offsetIsFP) {
319  // Handle rowwise quantization for Rowwise quantized SLS.
320  float scale = ((double)max - (double)min) / 255.0;
321  float offset = min;
322 
323  for (size_t j = 0; j < idim.width; j++) {
324  destH.at({i, j}) = quantization::quantizeWithFloatOffset<QP>(
325  srcH.at({i, j}), scale, offset);
326  }
327  scalesH.raw(i) = static_cast<ScaleT>(scale);
328  offsetsH.raw(i) = static_cast<OffsetT>(offset);
329  } else {
330  llvm_unreachable("Unsupported offset type.");
331  }
332  }
333 }

355 template <typename T>
356 void tensorFusedRowwiseQuantization(const Tensor &input, Tensor &output) {
357  // We are fusing the scale and offset onto the end of each row. Thus input and
358  // output must both be 2 dimensional, with output having 2*sizeof(T) extra
359  // columns for the scale and offset.
360  auto outputType = output.getElementType();
361  assert(input.dims().size() == 2 && output.dims().size() == 2 &&
362  "Input and output must be 2 dimensional.");
363  if (outputType == ElemKind::UInt8FusedFP16QTy ||
364  outputType == ElemKind::UInt8FusedQTy) {
365  assert(input.dims()[1] + 2 * sizeof(T) == output.dims()[1] &&
366  "Output must have 2*sizeof(T) more columns than input for 8-bits "
367  "quantization.");
368  } else if (outputType == ElemKind::UInt4FusedFP16QTy) {
369  constexpr bool scaleIsFP16 = std::is_same<float16_t, T>::value;
370  (void)scaleIsFP16;
371  assert(scaleIsFP16 && "Only float16_t scale and offset are supported "
372  "in 4-bit fused quantization");
373  assert(
374  input.dims()[1] % 2 == 0 &&
375  "4-bits fused quantization only works for the number of input column "
376  "a multiple of 2");
377  assert(
378  input.dims()[1] / 2 + 2 * sizeof(T) == output.dims()[1] &&
379  "Output must have 2*sizeof(T) more columns than half of input columns "
380  "for 4-bits quantization.");
381  }
382 
383  auto srcH = input.getHandle<float>();
384  auto destH = output.getHandle<uint8_t>();
385  for (size_t i = 0, e = input.dims()[0]; i < e; i++) {
386  auto slice = srcH.extractSlice(i);
387  auto rSrc = slice.getHandle<float>();
388  auto res = rSrc.minMaxArg();
389  float min = rSrc.raw(res.first);
390  float max = rSrc.raw(res.second);
391 
392  min = std::min(min, 0.0f);
393  max = std::max(max, 0.0f);
394 
395  float range;
396  switch (outputType) {
397  case ElemKind::UInt8FusedQTy:
398  case ElemKind::UInt8FusedFP16QTy:
399  range = 255.0;
400  break;
401  case ElemKind::UInt4FusedFP16QTy:
402  range = 15.0;
403  break;
404  default:
405  llvm_unreachable("Not yet supported");
406  }
407 
408  // This matches the Caffe2 implementation for FloatToRowwiseQuantized8BitsOp
409  // found in operators/lengths_reducer_rowwise_8bit_ops.h.
410  constexpr float kEqualityThreshold = 1e-10f;
411  const float scale = ((max - min) < kEqualityThreshold)
412  ? 1.0
413  : ((double)max - (double)min) / range;
414  const float offset = min;
415 
416  for (size_t j = 0, f = input.dims()[1]; j < f; j++) {
417  if (outputType == ElemKind::UInt8FusedFP16QTy ||
418  outputType == ElemKind::UInt8FusedQTy) {
419  destH.at({i, j}) = quantization::quantizeWithFloatOffset<uint8_t>(
420  srcH.at({i, j}), scale, offset);
421  } else if (outputType == ElemKind::UInt4FusedFP16QTy) {
422  uint8_t quantized = quantization::quantize4BitsWithFloatOffset(
423  srcH.at({i, j}), scale, offset);
424  if (j % 2 == 0) {
425  // Even columns use LSB 4-bit.
426  destH.at({i, j / 2}) = quantized;
427  } else {
428  // Odd columns use MSB 4-bit.
429  destH.at({i, j / 2}) |= quantized << 4;
430  }
431  } else {
432  llvm_unreachable("Not yet supported");
433  }
434  }
435 
436  // Now set the scale/offset at the end of each row.
437  destH.setFusedScaleOffsetInRow<T>(i, scale, offset);
438  }
439 }
